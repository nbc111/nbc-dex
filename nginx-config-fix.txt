# ============================================
# NBC DEX Nginx 配置文件
# 文件路径: /etc/nginx/sites-enabled/99nbblock-dex
# 用途: 为 dex.nbblocks.cc 提供静态文件服务
# ============================================

# HTTP 重定向到 HTTPS
server {
    listen 80;
    server_name dex.nbblocks.cc www.dex.nbblocks.cc;
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl;
    http2 on;
    server_name dex.nbblocks.cc www.dex.nbblocks.cc;

    # SSL 证书配置
    ssl_certificate /etc/nginx/ssl/dex.nbblocks.cc.pem;
    ssl_certificate_key /etc/nginx/ssl/dex.nbblocks.cc.key;

    # SSL 协议和加密套件
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384:ECDHE-ECDSA-CHACHA20-POLY1305:ECDHE-RSA-CHACHA20-POLY1305:DHE-RSA-AES128-GCM-SHA256:DHE-RSA-AES256-GCM-SHA384;
    ssl_prefer_server_ciphers off;
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 10m;

    # 安全响应头
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;

    # 静态文件根目录
    root /www/nbc-dex/uniswap-interface-moonbeam/build;
    index index.html;

    # 启用 Gzip 压缩
    gzip on;
    gzip_vary on;
    gzip_min_length 1024;
    gzip_proxied any;
    gzip_comp_level 6;
    gzip_types text/plain text/css text/xml text/javascript application/json application/javascript application/xml+rss application/rss+xml font/truetype font/opentype application/vnd.ms-fontobject image/svg+xml;

    # 禁用 index.html 的缓存（必须在 location / 之前）
    location = /index.html {
        add_header Cache-Control "no-cache, no-store, must-revalidate";
        add_header Pragma "no-cache";
        add_header Expires "0";
        try_files $uri =404;
    }

    # 主 location - 支持 React Router
    location / {
        try_files $uri $uri/ /index.html;
    }

    # 静态资源缓存优化（JS, CSS, 图片等）
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2|ttf|eot)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
        access_log off;
    }

    # 静态资源目录（React 构建的 static 目录）
    location /static {
        expires 1y;
        add_header Cache-Control "public, immutable";
        access_log off;
    }

    # 禁止访问隐藏文件和敏感文件
    location ~ /\. {
        deny all;
        access_log off;
        log_not_found off;
    }

    # 禁止访问备份文件
    location ~ ~$ {
        deny all;
        access_log off;
        log_not_found off;
    }

    # 错误页面（可选）
    error_page 404 /index.html;

    # 日志配置
    access_log /var/log/nginx/dex.nbblocks.cc.access.log;
    error_log /var/log/nginx/dex.nbblocks.cc.error.log;
}
